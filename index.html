cls ; `
cd $env:SPLUNK_HOME\bin ; `
.\splunk.exe btool server list --debug | sls -Pattern "\\local\\" | sls -Pattern "(pass4SymmKey =)|(sslPassword =)"; `
echo "" ; `
.\splunk.exe btool server list --debug | sls -Pattern "\\default\\" | sls -Pattern "(pass4SymmKey =)|(sslPassword =)"; `
"-"*60 ; `
$the_file = (gci $env:SPLUNK_HOME\etc\system\default\ server.conf).FullName ; `
foreach ($x in $the_file){echo "" $x "" ""; Get-Content $x |  Select-String -Pattern '(pass4SymmKey =)|(sslPassword =)' ; echo "---"} ; `
echo "pass4SymmKey" ; `
.\splunk.exe show-decrypted --value $($(.\splunk.exe btool server list --debug | sls -Pattern "\\local\\" | sls -Pattern "pass4SymmKey") -split "pass4SymmKey = ")[1] ; `
echo "" "sslPassword" ; `
.\splunk.exe show-decrypted $($(.\splunk.exe btool server list --debug | sls -Pattern "\\local\\" | sls -Pattern "sslPassword") -split "sslPassword = ")[1]
